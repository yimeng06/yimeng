给你单链表的头结点 head ，请你找出并返回链表的中间结点。
如果有两个中间结点，则返回第二个中间结点。
快慢指针的一个入门题目。

class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        ListNode* slow=head;
        ListNode* fast=head;
        while(fast&&fast->next)
        {
            slow=slow->next;
            fast=fast->next->next;
        }
        return slow;
    }
};

接下来的是一个环形的问题
如果链表中存在环 ，则返回 true 。 否则，返回 false
class Solution {
public:
    bool hasCycle(ListNode *head) {
        ListNode* slow=head;
        ListNode* fast=head;
        while(fast&&fast->next)
        {
            slow=slow->next;
            fast=fast->next->next;
            if(fast==slow)
            {
                return true;
            }
        }
        return false;
    }
};

接下来这个是要返回环的入口节点
原理如下：
假设快慢指针在环内相遇走了b步，那么快指针相当于走了2b步，慢指针走了b步，设从头节点走到入环口需要走a步，再假设环长为c，快慢相遇时块指针比慢指针多走了kc圈，得2b-b=kc,b=kc，此时再让头节点head开始位移同时慢指针也开始位移，在此时慢指针在环内已经走了b-a步，b-a等于kc-a,那么就说明慢指针再走a步就回到了入口节点，如果用C++表示那就是在快慢指针相遇后，快指针就不管他了，让慢指针和head开始移动，当他们相遇的地方就是入口节点，也就是都走了a步
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode* slow=head;
        ListNode* fast=head;
        while(fast&&fast->next)
        {
            slow=slow->next;
            fast=fast->next->next;
            if(fast==slow)
            {
                while(slow!=head)
                {
                    head=head->next;
                    slow=slow->next;
                }
                return slow;
            }
        }
        return nullptr;
    }
};

重排链表
L0 → L1 → … → Ln - 1 → Ln
请将其重新排列后变为：
L0 → Ln → L1 → Ln - 1 → L2 → Ln - 2 → …

class Solution {
    //链表中间值
    ListNode* middleNode(ListNode* head)
    {
        ListNode* slow=head,*fast=head;
        while(fast&&fast->next)
        {
            slow=slow->next;
            fast=fast->next->next;
        }
        return slow;
    }
    //反转链表
    ListNode* reverseList(ListNode* head)
    {
        ListNode* pre=nullptr,*cur=head;
        while(cur)
        {
            ListNode* nxt=cur->next;
            cur->next=pre;
            pre=cur;
            cur=nxt;
        }
        return pre;
    }
public:
    void reorderList(ListNode* head) {
        ListNode* mid=middleNode(head);
        ListNode* head2=reverseList(mid);
        while(head2->next)//中间的那个指向为nullptr
        {
            ListNode* nxt=head->next;//记录下一个节点
            ListNode* nxt2=head2->next;//记录下一个节点
            head->next=head2;//连接链表头连尾
            head2->next=nxt;//尾连接到下一个节点
            head=nxt;
            head2=nxt2;
        }
    }
};
