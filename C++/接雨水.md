class Solution {
public:
    int trap(vector<int>& height) {
        int n = height.size();
        vector<int> pre_max(n); // pre_max[i] 表示从 height[0] 到 height[i] 的最大值
        pre_max[0] = height[0];
        for (int i = 1; i < n; i++) {
            pre_max[i] = max(pre_max[i - 1], height[i]);
        }

        vector<int> suf_max(n); // suf_max[i] 表示从 height[i] 到 height[n-1] 的最大值
        suf_max[n - 1] = height[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            suf_max[i] = max(suf_max[i + 1], height[i]);
        }

        int ans = 0;
        for (int i = 0; i < n; i++) {
            ans += min(pre_max[i], suf_max[i]) - height[i]; // 累加每个水桶能接多少水
        }//在前缀和后缀之中选最小值，减去当前位置的高度，就是当前位置能接多少水
        return ans;
    }
};

还有进阶写法可以节省空间复杂度（相向双指针）
每个位置的雨水量由它左右两侧最高柱子中的较矮者决定。
因此，我们可以维护两个指针，一个指向左侧最高的柱子，一个指向右侧最高的柱子，
并计算当前位置的雨水量。如果左侧最高的柱子比右侧最高的柱子矮，
则当前位置的雨水量等于左侧最高的柱子减去当前位置的高度，（记住是当前位置）！！！
并将左侧指针向右移；否则，当前位置的雨水量等于右侧最高的柱子减去当前位置的高度，
并将右侧指针向左移。
class Solution {
public:
    int trap(vector<int>& height) {
        int ans = 0, pre_max = 0, suf_max = 0;
        int left = 0, right = height.size() - 1;
        while (left < right) {
            pre_max = max(pre_max, height[left]);//更新左侧最高的柱子
            suf_max = max(suf_max, height[right]);//更新右侧最高的柱子
            ans += pre_max < suf_max ? pre_max - height[left++] : suf_max - height[right--];//计算当前位置的雨水量
        }
        return ans;
    }
};
