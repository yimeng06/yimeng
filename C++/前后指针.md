class Solution {
public:
    void deleteNode(ListNode* node) {
        *node = *node->next;
    }
};
这是一个面试题，重点考察对链表操作的理解
给定节点的值不应该存在于链表中。
链表中的节点数应该减少 1。
node 前面的所有值顺序相同。
node 后面的所有值顺序相同。
并且不能知晓前面的值，但是在实际引用中C++要加上delete关键字，否则会出现内存泄漏。如下：
class Solution {
public:
    void deleteNode(ListNode* node) {
        auto nxt = node->next;
        *node = *nxt;
        delete nxt;
    }
};


给你一个链表，删除链表的倒数第 n 个结点，并且返回链表的头结点。

class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        // 由于可能会删除链表头部，用哨兵节点简化代码
        ListNode dummy{0, head};
        ListNode* left = &dummy;
        ListNode* right = &dummy;
        while (n--) {
            right = right->next; // 右指针先向右走 n 步
        }
        while (right->next) {
            left = left->next;
            right = right->next; // 左右指针一起走
        } //那么left和right永远相差n个节点，并且当right走到链表尾部时，left就在倒数第n个节点的前一个节点也就是n+1节点处
        // 左指针的下一个节点就是倒数第 n 个节点
        ListNode* nxt = left->next;
        left->next = left->next->next;
        delete nxt;//防止内存泄漏
        return dummy.next;
    }
};

查重问题，有重复的就删掉，例如1->1->2->3->3,等于1->2->3

class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if(head==nullptr)
        {
            return nullptr;
        }
        ListNode* cur=head;
        while(cur->next)//看下一个节点
        {
            if(cur->next->val==cur->val)//和我一样，滚！
            {
                cur->next=cur->next->next;
            }else{//和我不一样，继续！
                cur=cur->next;
            }
        }
        return head;
    }
};

class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        ListNode dummy(0,head);
        auto cur = &dummy;
        while(cur->next&&cur->next->next)
        {
            int val=cur->next->val;
            if(cur->next->next->val==val)
            {//后两个节点值相同
            //值等于val的节点都删掉
                while(cur->next&&cur->next->val==val)
                {
                    cur->next=cur->next->next;
                }
            }else{
                cur=cur->next;
            }
        }
        return dummy.next;
    }
};
