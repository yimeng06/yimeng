~ 符号表示这是一个析构函数
析构函数在对象被销毁时自动调用，用于清理对象占用的资源
~Person() {
    delete[] name;
}
常常和new运算符一起使用
class Person {
public:
    char* name;  // 指向动态分配的内存
    
    Person(const char* name, int age) {
        this->name = new char[strlen(name) + 1];  // 动态分配
        // ...
    }
    
    ~Person() {
        if (name != nullptr) {
            delete[] name;      // 释放内存
            name = nullptr;     // 防止悬空指针
//在C++中，nullptr 是空指针常量，表示指针不指向任何对象。
        }
    }
}
