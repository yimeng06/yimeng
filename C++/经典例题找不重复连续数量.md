#include <iostream>
#include <string>
#include <unordered_set>
//哈希表有两个，unordered_map：存储键值对（key-value pairs），其中键（key）和值（value）可以是任意类型，不一定是int。他的头文件是<unordered_map>。
//unordered_set：只有key，不存储value，只存储key，而且key是唯一的，不允许重复。他的头文件是<unordered_set>。

using namespace std;

// 你的 Solution 类放在这里
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        unordered_set<char> occ; // 哈希集合，记录每个字符是否出现过
        int n = s.size();
        int rk = -1, ans = 0;  // 右指针，初始值为 -1，相当于我们在字符串的左边界的左侧，还没有开始移动
        for (int i = 0; i < n; ++i) {
            if (i != 0) {
                occ.erase(s[i - 1]); // 左指针向右移动一格，移除一个字符
            }
            while (rk + 1 < n && !occ.count(s[rk + 1])) {
                occ.insert(s[rk + 1]);
                ++rk; // 不断地移动右指针
            }
            ans = max(ans, rk - i + 1); // 第 i 到 rk 个字符是一个极长的无重复字符子串
        }
        return ans;
    }
};

int main() {
    Solution solution;  // 创建 Solution 对象
    
    // 测试用例
    string test1 = "abcabcbb";
    string test2 = "bbbbb";
    string test3 = "pwwkew";
    string test4 = "abc";
    
    cout << "测试 '" << test1 << "': " << solution.lengthOfLongestSubstring(test1) << endl;
    cout << "测试 '" << test2 << "': " << solution.lengthOfLongestSubstring(test2) << endl;
    cout << "测试 '" << test3 << "': " << solution.lengthOfLongestSubstring(test3) << endl;
    cout << "测试 '" << test4 << "': " << solution.lengthOfLongestSubstring(test4) << endl;
    
    // 你也可以让用户输入字符串
    string userInput;
    cout << "请输入一个字符串: ";
    cin >> userInput;
    cout << "最长无重复字符子串长度: " << solution.lengthOfLongestSubstring(userInput) << endl;
    
    return 0;
}
