请你返回所有和为 0 且不重复的三元组。

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        ranges::sort(nums); // 排序
        vector<vector<int>> ans;
        int n = nums.size();
        for(int i=0;i<n-2;i++)
        {
            int x = nums[i];
            if(i&&x==nums[i-1]) continue;//跳过重复的数字
            if(x+nums[i+1]+nums[i+2]>0) break;//如果x太大，后面的数字都太小，肯定没戏
            if(x+nums[n-1]+nums[n-2]<0) continue;//如果x太小，后面的数字都太大，肯定没戏，但还能变大所以是continue
            int j=i+1,k=n-1;
            while(j<k)
            {
                int s= x+nums[j]+nums[k];
                if(s>0)
                {
                    k--;
                }
                else if(s<0)
                {
                    j++;
                }
                else{ //三数之和为0
                    ans.push_back({x,nums[j],nums[k]});
                    for(j++;j<k&&nums[j]==nums[j-1];j++);//跳过重复的数字
                    for(k--;k>j&&nums[k]==nums[k+1];k--);//跳过重复的数字
                }
            }
        }
        return ans;
    }
};

补充：ranges::sort(nums) 是C++20引入的排序算法库，可以代替sort()函数。
头文件为<algorithm>。
可以节省代码量。
例如：原始的sort()函数：
sort(nums.begin(), nums.end());
但使用ranges可以简化为：    ranges::sort(nums);//从大到小


两数之和（有序排列）
class Solution {
public:
    vector<int> twoSum(vector<int>& numbers, int target) {
        int left = 0,right = numbers.size()-1;
        while(1)
        {
            int s = numbers[left]+numbers[right];
            if(s==target)
            {
                return {left+1,right+1};
            }
            s>target ? right--:left++;
        }
    }
};
运用双指针法，先固定一个数字，然后在另一个数组中寻找另一个数字，使得两数字之和等于目标值。
