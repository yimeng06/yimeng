#include <iostream>
#include <iomanip> //这个头文件作用是控制输入输出格式化，也就是这个函数fixed<<setprecision(1)，输出保留一个小数
#include <vector>
#include <algorithm>

using namespace std;

class Employee {

    private:
        string name;
        double salary;
    
    public:
        //构造函数
        Employee(string name,double salary){
            this->name=name;
            this->salary=salary;
        }
    
        //获取名字
        string getName(){
            return name;
        }
    
        //获取薪水
        double getSalary(){
            return salary;
        }

};

//重载比较，按薪水从大到小
bool cmp(Employee& e1,Employee& e2){
    return e1.getSalary()>e2.getSalary();
} //bool其返回值的含义返回 true：表示 e1 应该排在 e2 前面
                        返回 false：表示 e1 应该排在 e2 后面

void print(Employee& e){
    //个人所得税
    double tax=0.0;
    //全月应纳税所得额
    double t=e.getSalary()-3500;
    //按表中所给的范围，分别进行计算
    if(t<=0){
        //小于0，不需要扣税
    }
    //其它情况，按所给公式及表中数据计算
    else if(t>0&&t<=1500){
        tax=t*0.03-0;
    }
    else if(t>1500&&t<=4500){
        tax=t*0.10-105;
    }
    else if(t>4500&&t<=9000){
        tax=t*0.20-555;
    }
    else if(t>9000&&t<=35000){
        tax=t*0.25-1005;
    }
    else if(t>35000&&t<=55000){
        tax=t*0.30-2755;
    }
    else if(t>55000&&t<=80000){
        tax=t*0.35-5505;
    }
    else{
        tax=t*0.45-13505;
    }
    //设置精度
    cout<<fixed<<setprecision(1);
    cout<<e.getName()<<"应该缴纳的个人所得税是："<<tax<<endl;
}

int main() {

    //新建三个Employee对象
    Employee e1("张三",6500);
    Employee e2("李四",8000);
    Employee e3("王五",100000);
    vector<Employee> vec;
    //添加到容器
    vec.push_back(e1);
    vec.push_back(e2);
    vec.push_back(e3);
    //按工资从大到小排序
    sort(vec.begin(),vec.end(),cmp); //sort里面的cmp的作用是给予规则
    //输出个人所得税
    for_each(vec.begin(),vec.end(),print);  //是<algorithm> 头文件中的函数，定义功能为对容器中每个元素执行指定的操作，这里的作用是对vec中的每个元素调用print函数

    return 0;
}

输出结果：
王五应该缴纳的个人所得税是：32000.0
李四应该缴纳的个人所得税是：1350.0
张三应该缴纳的个人所得税是：0.0
