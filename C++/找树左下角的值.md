非常简单的一个思路和写法
BFS 这棵二叉树，先把右儿子入队，再把左儿子入队，这样最后一个出队的节点就是左下角的节点了。

class Solution {
public:
    int findBottomLeftValue(TreeNode* root) {
        TreeNode* node;
        queue<TreeNode*> q;
        q.push(root);
        while(!q.empty())
        {
            node = q.front();
            q.pop();
            if(node->right) q.push(node->right);
            if(node->left)  q.push(node->left);
        }
        return node->val;
    }
};
