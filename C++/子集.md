输入：nums = [1,2,3]
输出：[[],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3]]

这是一个二进制枚举解法
class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
        int n=nums.size();
        vector<vector<int>> ans(1<<n);
        for(int i = 0;i<(1<<n);i++){
            for(int j=0;j<n;j++){
                if(i>>j&1){
                    ans[i].push_back(nums[j]);
                }
            }
        }
        return ans;
    }
};

其原理如下
n = 3
ans 大小 = 1 << 3 = 8

循环 i 从 0 到 7：

i = 0 (二进制: 000)
  j=0: (0 >> 0 & 1) = 0 → 不选
  j=1: (0 >> 1 & 1) = 0 → 不选
  j=2: (0 >> 2 & 1) = 0 → 不选
  ans[0] = []

i = 1 (二进制: 001)
  j=0: (1 >> 0 & 1) = 1 → 选 nums[0]=1
  j=1: (1 >> 1 & 1) = 0 → 不选
  j=2: (1 >> 2 & 1) = 0 → 不选
  ans[1] = [1]

i = 2 (二进制: 010)
  j=0: (2 >> 0 & 1) = 0 → 不选
  j=1: (2 >> 1 & 1) = 1 → 选 nums[1]=2
  j=2: (2 >> 2 & 1) = 0 → 不选
  ans[2] = [2]

i = 3 (二进制: 011)
  j=0: (3 >> 0 & 1) = 1 → 选 nums[0]=1
  j=1: (3 >> 1 & 1) = 1 → 选 nums[1]=2
  j=2: (3 >> 2 & 1) = 0 → 不选
  ans[3] = [1, 2]

i = 4 (二进制: 100)
  j=0: (4 >> 0 & 1) = 0 → 不选
  j=1: (4 >> 1 & 1) = 0 → 不选
  j=2: (4 >> 2 & 1) = 1 → 选 nums[2]=3
  ans[4] = [3]

i = 5 (二进制: 101)
  j=0: (5 >> 0 & 1) = 1 → 选 nums[0]=1
  j=1: (5 >> 1 & 1) = 0 → 不选
  j=2: (5 >> 2 & 1) = 1 → 选 nums[2]=3
  ans[5] = [1, 3]

i = 6 (二进制: 110)
  j=0: (6 >> 0 & 1) = 0 → 不选
  j=1: (6 >> 1 & 1) = 1 → 选 nums[1]=2
  j=2: (6 >> 2 & 1) = 1 → 选 nums[2]=3
  ans[6] = [2, 3]

i = 7 (二进制: 111)
  j=0: (7 >> 0 & 1) = 1 → 选 nums[0]=1
  j=1: (7 >> 1 & 1) = 1 → 选 nums[1]=2
  j=2: (7 >> 2 & 1) = 1 → 选 nums[2]=3
  ans[7] = [1, 2, 3]
