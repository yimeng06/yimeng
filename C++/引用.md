命名空间(::为作用域运算符)
格式：
namespace namespace_name {
     int varl;
     void funcl();
     namespace sub_namespace{
            //namespace content
}
}

运用：
#include <iostream>
using namespace std;

namespace yimu
{
	int age = 18;
	void fun()
	{
		cout << "aga=" << age << endl;
	}
	namespace sun_yimu
	{
		void fun()
		{
			cout << "sun_yimu函数" << endl;
		}
	}
}

int main() {
	yimu::fun();//：：运用域运算符
	yimu::sun_yimu::fun();//也可以嵌套使用
	system("pause");
	return 0;
}
