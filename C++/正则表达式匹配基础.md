给你一个字符串 s 和一个字符规律 p，请你来实现一个支持 '.' 和 '*' 的正则表达式匹配。

'.' 匹配任意单个字符
'*' 匹配零个或多个前面的那一个元素
所谓匹配，是要涵盖 整个 字符串 s 的，而不是部分字符串。

class Solution {
public:
    bool isMatch(string s, string p) {
        int m = s.size()+1,n = p.size()+1; // 多加一个空格
        vector<vector<bool>>dp(m,vector<bool>(n,false));//创建二维布尔数组
        dp[0][0] = true; // 空字符串匹配空字符串
        for(int j = 2;j<n;j+=2)
        {
            dp[0][j]=dp[0][j - 2]&&p[j-1]=='*';
        }
        for(int i =1;i<m;i++)
        {
            for(int j =1;j<n;j++)
            {
                dp[i][j]=p[j-1]=='*'? // 处理 '*' 匹配 
                dp[i][j-2]||dp[i-1][j]&&(s[i-1]==p[j-2]||p[j-2]=='.'):
                dp[i-1][j-1]&&(p[j-1]=='.'||s[i-1]==p[j-1]);
            }
        }
        return dp[m-1][n-1];
    }
};
