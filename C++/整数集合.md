set<int> s;

// 添加元素
s.insert(10);
s.insert(20);
s.insert(30);

// 删除元素
s.erase(20);

// 检查元素是否存在
if (s.count(10)) {
    cout << "10在集合中";
}

// 遍历集合
for (auto num : s) {
    cout << num << " ";  // 输出：10 30
}

// 获取大小
cout << s.size();  // 输出：2

他主要有三个作用
1. 自动去重
set<int> s;
s.insert(1);
s.insert(2);
s.insert(1);  // 这个1不会被重复添加
s.insert(3);

// 集合s中只有：1, 2, 3
2. 自动排序
set<int> s;
s.insert(5);
s.insert(1);
s.insert(3);

// 集合s会自动排序为：1, 3, 5
3. 快速查找
set<int> s = {1, 2, 3, 4, 5};
if (s.find(3) != s.end()) {
    cout << "3在集合中";  // 会输出这个
}

s.upper_bound
它的作用是在集合 s 中查找第一个大于给定值的元素的迭代器，如果集合中没有大于给定值的元素，则返回 s.end()，表示超出集合的末尾。
set<int> s = {1, 3, 5, 7, 9};

auto it = s.upper_bound(3);  // it 指向 5
auto it2 = s.upper_bound(5); // it2 指向 7
auto it3 = s.upper_bound(10); // it3 指向 s.end()，因为没有大于 10 的元素
s.lower_bound
它的作用是在集合 s 中查找第一个大于等于给定值的元素的迭代器，如果集合中没有大于等于给定值的元素，则返回 s.end()，表示超出集合的末尾。
set<int> s = {1, 3, 5, 7, 9};

auto it = s.lower_bound(3);  // it 指向 3
auto it2 = s.lower_bound(5); // it2 指向 5
auto it3 = s.lower_bound(10); // it3 指向 s.end()，因为没有大于等于 10 的元素
