力扣17. 电话号码的字母组合

dfs(i) 处理的是从下标 i 到末尾的所有字母组合。dfs(i) 不仅仅在枚举 i，还包含了 dfs(i+1),dfs(i+2),…,dfs(n) 这之后的所有递归调用。单纯说「枚举 i」是不准确的，因为除了枚举 i，还要递归处理剩余部分。正因为如此，我视频中讲的是 ≥i 而不是等于 i。

class Solution {
    static constexpr string MAPPING[10] = {"", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};

public:
    vector<string> letterCombinations(string digits) {
        int n = digits.length();
        if (n == 0) {
            return {};
        }

        vector<string> ans;
        string path(n, 0);  // 创建长度为n的字符串，初始值为0（空字符
        // 注意 path 长度一开始就是 n，不是空串

        // lambda 递归
        auto dfs = [&](this auto&& dfs, int i) -> void {
            if (i == n) {
                ans.emplace_back(path);// 将完整组合加入结果集
                return;
            }
            for (char c : MAPPING[digits[i] - '0']) {
                path[i] = c; // 直接覆盖// 直接修改第i个位置的字符
                dfs(i + 1);
            }
        };

        dfs(0); //启动回溯递归
        return ans;
    }
};


其中
ans.push_back(path);     // 拷贝构造：先创建临时对象，再拷贝
ans.emplace_back(path);  // 直接构造：在容器内直接构造对象
